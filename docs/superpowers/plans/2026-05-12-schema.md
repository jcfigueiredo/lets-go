**Goal:** Land the lab experiment tracking schema in 9 iterations — one vertical slice per entity. Each iteration produces a model + migration + factory + idempotent seed extension + tests, with the suite staying green at 100% branch coverage throughout.

**Architecture:** SQLModel classes mirror the SQL in [the schema design](../specs/2026-05-12-schema-design.md). Alembic autogenerates migrations from `lab.models`; CHECK constraints and edge cases get hand-tidied. Tests use factory-boy with the existing SAVEPOINT-isolated `db` fixture. Seed is hand-curated, deterministic, and idempotent via `INSERT … ON CONFLICT DO NOTHING`.

**Tech Stack:** SQLModel + SQLAlchemy 2.x (sync) + psycopg + Alembic + pytest + factory-boy + Faker + ruff. All landed in the infra plan; this plan only adds factory-boy and faker.

```python
"""Researcher aggregate root.

A scientist who runs experiments. Has identity, contact, and a lab-global role.
"""

from datetime import datetime
from enum import StrEnum

from sqlalchemy import Column, Enum as SAEnum, func
from sqlmodel import Field, SQLModel


class ResearcherRole(StrEnum):
    PRINCIPAL_INVESTIGATOR = "principal_investigator"
    LAB_TECHNICIAN = "lab_technician"
    GRADUATE_STUDENT = "graduate_student"
    POSTDOC = "postdoc"
    UNDERGRADUATE = "undergraduate"


class Researcher(SQLModel, table=True):
    __tablename__ = "researchers"

    id: int | None = Field(default=None, primary_key=True)
    name: str = Field(nullable=False)
    email: str = Field(nullable=False, unique=True)
    role: ResearcherRole = Field(
        sa_column=Column(SAEnum(ResearcherRole, name="researcher_role"), nullable=False),
    )
    created_at: datetime = Field(
        sa_column_kwargs={"server_default": func.now()},
        nullable=False,
    )
    updated_at: datetime = Field(
        sa_column_kwargs={"server_default": func.now(), "onupdate": func.now()},
        nullable=False,
    )
```

> **Why explicit `sa_column=Column(SAEnum(...))`:** the default SQLModel→SQLAlchemy enum mapping derives the Postgres type name from the Python class — `ResearcherRole` becomes `researcherrole` (no underscores), which doesn't match the design spec's `researcher_role`. The explicit form fixes the name. This pattern repeats for every other enum-backed column (Tasks 2, 5, 7).

```python
"""Seed entry point.

Hand-curated, deterministic data exercising the spec scenarios. Idempotent:
``make seed && make seed`` produces the same final state.

Idempotency strategy: ``INSERT … ON CONFLICT DO NOTHING`` on the natural
identifier (email, accession code, composite PK, etc.).
"""

from sqlalchemy.dialects.postgresql import insert
from sqlmodel import Session

from lab.db import engine
from lab.models import Researcher, ResearcherRole


def _seed_researchers(session: Session) -> None:
    data = [
        {"name": "Alice Tan",   "email": "alice@lab.example",   "role": ResearcherRole.PRINCIPAL_INVESTIGATOR},
        {"name": "Bob Singh",   "email": "bob@lab.example",     "role": ResearcherRole.LAB_TECHNICIAN},
        {"name": "Carol Liu",   "email": "carol@lab.example",   "role": ResearcherRole.GRADUATE_STUDENT},
        {"name": "Diego Cruz",  "email": "diego@lab.example",   "role": ResearcherRole.POSTDOC},
    ]
    stmt = insert(Researcher).values(data).on_conflict_do_nothing(index_elements=["email"])
    session.execute(stmt)


def seed(session: Session) -> None:
    """Apply seed data idempotently."""
    _seed_researchers(session)


if __name__ == "__main__":  # pragma: no cover
    with Session(engine) as session:
        seed(session)
        session.commit()
```

```python
"""Seed entry point. See docstring in Task 1's revision."""

from sqlalchemy.dialects.postgresql import insert
from sqlmodel import Session

from lab.db import engine
from lab.models import Project, ProjectStatus, Researcher, ResearcherRole


def _seed_researchers(session: Session) -> None:
    data = [
        {"name": "Alice Tan",  "email": "alice@lab.example",  "role": ResearcherRole.PRINCIPAL_INVESTIGATOR},
        {"name": "Bob Singh",  "email": "bob@lab.example",    "role": ResearcherRole.LAB_TECHNICIAN},
        {"name": "Carol Liu",  "email": "carol@lab.example",  "role": ResearcherRole.GRADUATE_STUDENT},
        {"name": "Diego Cruz", "email": "diego@lab.example",  "role": ResearcherRole.POSTDOC},
    ]
    stmt = insert(Researcher).values(data).on_conflict_do_nothing(index_elements=["email"])
    session.execute(stmt)


def _seed_projects(session: Session) -> None:
    # No UNIQUE constraint on title — use check-then-insert for idempotency.
    seeds = [
        {"title": "Glucose Tolerance Study",  "description": "Long-term tracking of glucose responses.", "status": ProjectStatus.ACTIVE},
        {"title": "Soil Microbiome Survey",   "description": "Comparative survey across three watersheds.", "status": ProjectStatus.PLANNING},
    ]
    from sqlmodel import select  # local import to keep module imports tidy
    for row in seeds:
        existing = session.exec(select(Project).where(Project.title == row["title"])).first()
        if existing is None:
            session.add(Project(**row))


def seed(session: Session) -> None:
    _seed_researchers(session)
    _seed_projects(session)


if __name__ == "__main__":  # pragma: no cover
    with Session(engine) as session:
        seed(session)
        session.commit()
```

```python
"""Measurement aggregate (per D1 + D3).

Polymorphic via discriminator + CHECK constraint:
  kind = 'numeric'     → numeric_value + unit populated; categorical_value + text_value NULL
  kind = 'categorical' → categorical_value populated; others NULL
  kind = 'text'        → text_value populated; others NULL
"""

from datetime import datetime
from decimal import Decimal
from enum import StrEnum

from sqlalchemy import CheckConstraint, Column, Enum as SAEnum
from sqlmodel import Field, SQLModel


class MeasurementKind(StrEnum):
    NUMERIC = "numeric"
    CATEGORICAL = "categorical"
    TEXT = "text"


class Measurement(SQLModel, table=True):
    __tablename__ = "measurements"
    __table_args__ = (
        CheckConstraint(
            """(
                (kind = 'numeric'     AND numeric_value IS NOT NULL AND unit IS NOT NULL
                                      AND categorical_value IS NULL AND text_value IS NULL)
             OR (kind = 'categorical' AND categorical_value IS NOT NULL
                                      AND numeric_value IS NULL AND unit IS NULL AND text_value IS NULL)
             OR (kind = 'text'        AND text_value IS NOT NULL
                                      AND numeric_value IS NULL AND unit IS NULL AND categorical_value IS NULL)
            )""",
            name="measurement_value_matches_kind",
        ),
    )

    id: int | None = Field(default=None, primary_key=True)
    experiment_id: int = Field(foreign_key="experiments.id", nullable=False)
    sample_id: int | None = Field(default=None, foreign_key="samples.id", nullable=True)
    recorded_by: int = Field(foreign_key="researchers.id", nullable=False)
    recorded_at: datetime = Field(nullable=False)
    kind: MeasurementKind = Field(
        sa_column=Column(SAEnum(MeasurementKind, name="measurement_kind"), nullable=False),
    )
    numeric_value: Decimal | None = Field(default=None, nullable=True)
    unit: str | None = Field(default=None, nullable=True)
    categorical_value: str | None = Field(default=None, nullable=True)
    text_value: str | None = Field(default=None, nullable=True)
    notes: str | None = Field(default=None, nullable=True)
```

```markdown
- **Don't break seed idempotency.** Every seed contribution uses either `INSERT … ON CONFLICT DO NOTHING` (when there's a UNIQUE constraint) or a check-then-insert pattern in Python. `make seed && make seed` must produce the same final state. - **Don't add a `relationship()` declaration on models without a concrete need.** The aggregates intentionally lack SQLModel `Relationship` attributes — queries use explicit `select().join()` instead.
```
